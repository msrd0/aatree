2use core::mem;

4impl<T: Ord> AANode<T> {
5	/// Insert a new node with `content` into the tree. If a node with this value already exist,
6	/// nothing will be inserted, and `false` will be returned.
7	pub fn insert(&mut self, content: T) -> bool {
8		let inserted = self.bst_insert(content);
9		if inserted {
10			let mut node = self.take();
11			node = node.skew().split();
12			*self = node;
13		}
14		inserted
15	}
16
17	/// Simple unbalanced BST insert.
18	fn bst_insert(&mut self, new: T) -> bool {
19		match self.as_mut() {
20			None => {
21				*self = new.into();
22				true
23			},
24			Some(Node {
25				content,
26				left_child,
27				..
28			}) if &new < content => left_child.insert(new),
29			Some(Node {
30				content,
31				right_child,
32				..
33			}) if &new > content => right_child.insert(new),
34			_ => false
35		}
36	}
37
38	/// Insert a new node with `content` into the tree. If a node with this value already exists,
39	/// it will be replaced and the old content returned.
40	pub fn insert_or_replace(&mut self, content: T) -> Option<T> {
41		let inserted = self.bst_insert_or_replace(content);
42		if inserted.is_none() {
43			let mut node = self.take();
44			node = node.skew().split();
45			*self = node;
46		}
47		inserted
48	}
49
50	/// Simple unbalanced BST insert or replace.
51	fn bst_insert_or_replace(&mut self, new: T) -> Option<T> {
52		match self.as_mut() {
53			None => {
54				*self = new.into();
55				None
56			},
57			Some(Node { content, .. }) if &new == content => {
58				Some(mem::replace(content, new))
59			},
60			Some(Node {
61				content,
62				left_child,
63				..
64			}) if &new < content => left_child.insert_or_replace(new),
65			Some(Node {
66				content,
67				right_child,
68				..
69			}) if &new > content => right_child.insert_or_replace(new),
70			_ => unreachable!()
71		}
72	}
