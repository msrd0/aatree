4/// This type specifies the requested step for [`traverse`](AANode::traverse).
5#[derive(Debug)]
6pub enum TraverseStep<R> {
7	Left,
8	Right,
9	Value(Option<R>)
10}
11
12impl<T> AANode<T> {
13	/// Traverse the tree looking for a specific value.
14	///
15	/// `down_callback` is called for each node on the way down the tree. It is passed the
16	/// value contained in the current node and may return either `Left` or `Right` to
17	/// continue the traversal in that direction, or `Value` to stop the traversal, for
18	/// example because a value was found.
19	///
20	/// `up_callback` is called while going back up with the content of each node and the
21	/// result of traversing so far (i.e., `None` for the first call when the search hit a
22	/// leaf, or the return value of the last callback execution otherwise).
23	pub fn traverse<'a, F, G, R>(&'a self, down_callback: F, up_callback: G) -> Option<R>
24	where
25		F: Fn(&'a T) -> TraverseStep<R> + Copy,
26		G: Fn(&'a T, Option<R>) -> Option<R> + Copy
27	{
28		self.as_ref().and_then(
29			|Node {
30			     content,
31			     left_child,
32			     right_child,
33			     ..
34			 }| {
35				let child = match down_callback(content) {
36					TraverseStep::Left => left_child,
37					TraverseStep::Right => right_child,
38					TraverseStep::Value(v) => return v
39				};
40				up_callback(content, child.traverse(down_callback, up_callback))
41			}
42		)
43	}
44}

48impl Display for TraverseMutError {
49	fn fmt(&self, f: &mut Formatter<'_>) -> fmt::Result {
50		write!(f, "Attempt to turn {} but there is no such child", self.0)
51	}

65impl<'a, T> TraverseMut<'a, T> {
66	/// Return if this node is a leaf (i.e. it is not nil and has no children).
67	pub(crate) fn is_leaf(&self) -> bool {
68		self.node.is_leaf()
69	}
70
71	/// Peek the content of this node (unless it is nil).
72	pub(crate) fn peek(&self) -> &T {
73		&self
74			.node
75			.as_ref()
76			.expect("This node should not be nil")
77			.content
78	}
79
80	/// Return the content of this node (unless it is nil).
81	pub(crate) fn into_content(self) -> &'a mut T {
82		&mut self
83			.node
84			.as_mut()
85			.expect("This node should not be nil")
86			.content
87	}
88
89	/// Return if this node has a left child.
90	pub(crate) fn has_left_child(&self) -> bool {
91		self.node
92			.as_ref()
93			.map(|node| !node.left_child.is_nil())
94			.unwrap_or(false)
95	}
96
97	/// Peek the content of the left child.
98	pub(crate) fn peek_left_child(&self) -> Option<&T> {
99		self.node
100			.as_ref()
101			.and_then(|node| node.left_child.as_ref().map(|left| &left.content))
102	}
103
104	/// Continue traversing the tree with the left child of the current node.
105	pub(crate) fn turn_left(self) -> Result<Self, TraverseMutError> {
106		Ok(Self {
107			node: self
108				.node
109				.as_mut()
110				.and_then(|node| {
111					(!node.left_child.is_nil()).then(|| &mut node.left_child)
112				})
113				.ok_or(TraverseMutError("left"))?
114		})
115	}
116
117	/// Return if this node has a right child.
118	pub(crate) fn has_right_child(&self) -> bool {
119		self.node
120			.as_ref()
121			.map(|node| !node.right_child.is_nil())
122			.unwrap_or(false)
123	}
124
125	/// Peek the content of the left child.
126	pub(crate) fn peek_right_child(&self) -> Option<&T> {
127		self.node
128			.as_ref()
129			.and_then(|node| node.right_child.as_ref().map(|left| &left.content))
130	}
131
132	/// Continue traversing the tree with the left child of the current node.
133	pub(crate) fn turn_right(self) -> Result<Self, TraverseMutError> {
134		Ok(Self {
135			node: self
136				.node
137				.as_mut()
138				.and_then(|node| {
139					(!node.right_child.is_nil()).then(|| &mut node.right_child)
140				})
141				.ok_or(TraverseMutError("right"))?
142		})
143	}
144}

146impl<T> AANode<T> {
147	/// Traverse the tree, allowing for mutation of the nodes that are being traversed.
148	///
149	/// **It is a logic error to mutate the nodes in a way that changes their order with
150	/// respect to the other nodes in the tree.**
151	pub(crate) fn traverse_mut(&mut self) -> Option<TraverseMut<'_, T>> {
152		(!self.is_nil()).then(|| TraverseMut { node: self })
153	}
